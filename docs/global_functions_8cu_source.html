23 // Copy original coordinates
24 dfloat3 proj = P;
25

27 // Update projection based on the direction of the normal vector
28 if (fabs(n.x - 1.0_df) < EPSILON) {
29 proj.x = 0.0_df;
30 } else if (fabs(n.x + 1.0_df) < EPSILON) {
31 proj.x = d;

33

35 proj.y = 0.0_df;

38 }
39

41 proj.z = 0.0_df;
42 } else if (fabs(n.z + 1.0_df) < EPSILON) {

97dfloat point_to_point_distance_periodic(dfloat3 p1, dfloat3 p2) {
98 dfloat3 delta = p1 - p2;
99
100 #ifdef BC_X_PERIODIC
101 if (delta.x > NX / 2.0_df) delta.x -= NX;
102 if (delta.x < -NX / 2.0_df) delta.x += NX;
103 #endif //BC_X_PERIODIC
104 #ifdef BC_Y_PERIODIC
105 if (delta.y > NY / 2.0_df) delta.y -= NY;
106 if (delta.y < -NY / 2.0_df) delta.y += NY;
107 #endif //BC_Y_PERIODIC
108 #ifdef BC_Z_PERIODIC
109 if (delta.z > NZ / 2.0_df) delta.z -= NZ;
110 if (delta.z < -NZ / 2.0_df) delta.z += NZ;
111 #endif //BC_Z_PERIODIC
112
113 return sqrtf(delta.x * delta.x + delta.y * delta.y + delta.z * delta.z);
114}

117__device__ dfloat3 getDiffPeriodic(const dfloat3& p1, const dfloat3& p2) {
118 dfloat dx, dy, dz;
119 // X direction
120 #ifdef BC_X_PERIODIC
121 dx = abs(p1.x - p2.x) > ((NX-1) / 2.0_df) ?
122 (p1.x < p2.x ? (p1.x + (NX-1) - p2.x) : (p1.x - (NX-1) - p2.x))
123 : p1.x - p2.x;
124 #else
125 dx = p1.x - p2.x;
126 #endif
127 // Y direction
128 #ifdef BC_Y_PERIODIC
129 dy = abs(p1.y - p2.y) > ((NY-1) / 2.0_df) ?
130 (p1.y < p2.y ? (p1.y + (NY-1) - p2.y) : (p1.y - (NY-1) - p2.y))
131 : p1.y - p2.y;
132 #else
133 dy = p1.y - p2.y;
134 #endif
135 // Z direction
136 #ifdef BC_Z_PERIODIC
137 dz = abs(p1.z - p2.z) > ((NZ-1) / 2.0_df) ?
138 (p1.z < p2.z ? (p1.z + (NZ-1) - p2.z) : (p1.z - (NZ-1) - p2.z))
139 : p1.z - p2.z;
140 #else
141 dz = p1.z - p2.z;
142 #endif
143 return dfloat3(dx, dy, dz);
144}

147dfloat point_to_segment_distance(dfloat3 p, dfloat3 segA, dfloat3 segB, dfloat3 closestOnAB[1]) {
148 dfloat3 ab = segB - segA;
149 dfloat3 ap = p - segA;
150 dfloat t = dot_product(ap, ab) / dot_product(ab, ab);
151 t = myMax(0, myMin(1, t)); // Clamp t to [0, 1]
152 closestOnAB[0] = segA + ab*t;
153 return vector_length(p - closestOnAB[0]);
154}

157dfloat point_to_segment_distance_periodic(dfloat3 p, dfloat3 segA, dfloat3 segB, dfloat3 closestOnAB[1]) {
158 dfloat minDist = 1E+37f;
159 dfloat3 bestClosestOnAB;
160
161 for (int i = 0; i < NUM_PERIODIC_DOMAIN_OFFSET; ++i) {
162 int dx = PERIODIC_DOMAIN_OFFSET[i][0];
163 int dy = PERIODIC_DOMAIN_OFFSET[i][1];
164 int dz = PERIODIC_DOMAIN_OFFSET[i][2];
165
166 // Translate the segment by the periodic offsets
167 dfloat3 segA_translated = segA + dfloat3(dx * NX, dy * NY, dz * NZ);
168 dfloat3 segB_translated = segB + dfloat3(dx * NX, dy * NY, dz * NZ);
169
170 // Compute the closest point on the translated segment
171 dfloat3 ab = segB_translated - segA_translated;
172 dfloat3 ap = p - segA_translated;
173
174 dfloat ab_dot_ab = dot_product(ab, ab);
175 dfloat t = 0.0f;
176 if (ab_dot_ab > 0) { //in case of zero-length segment
177 t = dot_product(ap, ab) / ab_dot_ab;
178 }
179
180 t = myMax(0.0f, myMin(1.0f, t)); // Clamp t to [0, 1]
181
182 dfloat3 tempClosestOnAB = segA_translated + ab * t;
183 dfloat dist = vector_length(p - tempClosestOnAB);
184
185 // Update the minimum distance and store the closest point
186 if (dist < minDist) {
187 minDist = dist;
188 bestClosestOnAB = tempClosestOnAB;
189 }
190 }
191
192 closestOnAB[0] = bestClosestOnAB;
193 return minDist;
194}

196// Project a point onto a segment and constrain it within the segment
197__device__
198dfloat3 constrain_to_segment(dfloat3 point, dfloat3 segStart, dfloat3 segEnd) {
199 dfloat3 segDir = segEnd - segStart;
200 dfloat segLengthSqr = dot_product(segDir,segDir);
201 if (segLengthSqr == 0.0_df)
202 return segStart; // The segment is a point
203
204 dfloat t = dot_product((point - segStart), segDir) / segLengthSqr;
205 t = clamp01(t);
206
207 return (segStart + (segDir * t));
208}

210// Main function to compute the closest distance between two segments and return the closest points
211__device__
212dfloat segment_segment_closest_points(dfloat3 p1, dfloat3 q1, dfloat3 p2, dfloat3 q2, dfloat3 closestOnAB[1], dfloat3 closestOnCD[1]) {
213
214 dfloat3 segDC = (q2 - p2); // Vector from p2 to q2 (segment [p2, q2])
215 dfloat lineDirSqrMag = dot_product(segDC,segDC); // Square magnitude of segment [p2, q2]
216
217 // Project p1 and q1 onto the plane defined by segment [p2, q2]
218 dfloat3 inPlaneA = p1 - ((dot_product(p1-p2,segDC)/lineDirSqrMag)*segDC);
219 dfloat3 inPlaneB = q1 - ((dot_product(q1-p2,segDC)/lineDirSqrMag)*segDC);
220 dfloat3 inPlaneBA = (inPlaneB - inPlaneA);
221 dfloat t = dot_product(p2-inPlaneA,inPlaneBA) / dot_product(inPlaneBA, inPlaneBA);
222
223
224 if (dot_product(inPlaneBA, inPlaneBA) == 0.0_df) {
225 t = 0.0_df; // Handle case where inPlaneA and inPlaneB are the same (segments are parallel)
226 }
227
228 // Find the closest point on segment [p1, q1] to the line [p2, q2]
229 dfloat3 segABtoLineCD = p1 + clamp01(t)*(q1-p1);
230
231 // Constrain the result to segment [p2, q2]
232 closestOnCD[0] = constrain_to_segment(segABtoLineCD, p2, q2);
233
234 // Constrain the closest point on segment [p2, q2] back to segment [p1, q1]
235 closestOnAB[0] = constrain_to_segment(closestOnCD[0], p1, q1);
236
237
238 // Calculate the distance between the closest points on the two segments
239 dfloat3 diff = vector_length(closestOnAB[0] - closestOnCD[0]);
240 return vector_length(diff); // Return the distance between the closest points
241}

316dfloat determinant_3x3(dfloat A[3][3]) {
317 return A[0][0] * (A[1][1] * A[2][2] - A[1][2] * A[2][1])
318 - A[0][1] * (A[1][0] * A[2][2] - A[1][2] * A[2][0])
319 + A[0][2] * (A[1][0] * A[2][1] - A[1][1] * A[2][0]);
320}

323void adjugate_3x3(dfloat A[3][3], dfloat adj[3][3]) {
324 adj[0][0] = A[1][1] * A[2][2] - A[1][2] * A[2][1];
325 adj[0][1] = -(A[1][0] * A[2][2] - A[1][2] * A[2][0]);
326 adj[0][2] = A[1][0] * A[2][1] - A[1][1] * A[2][0];
327
328 adj[1][0] = -(A[0][1] * A[2][2] - A[0][2] * A[2][1]);
329 adj[1][1] = A[0][0] * A[2][2] - A[0][2] * A[2][0];
330 adj[1][2] = -(A[0][0] * A[2][1] - A[0][1] * A[2][0]);
331
332 adj[2][0] = A[0][1] * A[1][2] - A[0][2] * A[1][1];
333 adj[2][1] = -(A[0][0] * A[1][2] - A[0][2] * A[1][0]);
334 adj[2][2] = A[0][0] * A[1][1] - A[0][1] * A[1][0];

338void inverse_3x3(dfloat A[3][3], dfloat result[3][3]) {
339 dfloat det = determinant_3x3(A);
340
341 dfloat adj[3][3];
342 adjugate_3x3(A, adj);
343

348 result[0][1] = adj[0][1] * inv_det;

378 dfloat4 q;
379
380 q.w = q1.w * q2.w - q1.x * q2.x - q1.y * q2.y - q1.z * q2.z;

432void quart_to_rotation_matrix(dfloat4 q, dfloat R[3][3]){
433 dfloat qx2 = q.x * q.x;
434 dfloat qy2 = q.y * q.y;
435 dfloat qz2 = q.z * q.z;
436 dfloat qwqx = q.w * q.x;
437 dfloat qwqy = q.w * q.y;
438 dfloat qwqz = q.w * q.z;
439 dfloat qxqy = q.x * q.y;
440 dfloat qxqz = q.x * q.z;
441 dfloat qyqz = q.y * q.z;
442
443 R[0][0] = 1 - 2 * (qy2 + qz2); R[1][0] = 2 * (qxqy - qwqz); R[2][0] = 2 * (qxqz + qwqy);
444 R[0][1] = 2 * (qxqy + qwqz); R[1][1] = 1 - 2 * (qx2 + qz2); R[2][1] = 2 * (qyqz - qwqx);
445 R[0][2] = 2 * (qxqz - qwqy); R[1][2] = 2 * (qyqz + qwqx); R[2][2] = 1 - 2 * (qx2 + qy2);
446}

450dfloat4 euler_to_quart(dfloat roll, dfloat pitch, dfloat yaw){
451 dfloat cr = cos(roll * 0.5_df);
452 dfloat sr = sin(roll * 0.5_df);
453 dfloat cp = cos(pitch * 0.5_df);
454 dfloat sp = sin(pitch * 0.5_df);
455 dfloat cy = cos(yaw * 0.5_df);
456 dfloat sy = sin(yaw * 0.5_df);
457
458 dfloat4 q;
459 q.w = cr * cp * cy + sr * sp * sy;
460 q.x = sr * cp * cy - cr * sp * sy;
461 q.y = cr * sp * cy + sr * cp * sy;
462 q.z = cr * cp * sy - sr * sp * cy;
463
464 return q;
465}

467__host__ __device__
468dfloat3 quart_to_euler(dfloat4 q){
469 dfloat3 angles;
470
471 // roll (x-axis rotation)
472 dfloat sinr_cosp = 2 * (q.w * q.x + q.y * q.z);
473 dfloat cosr_cosp = 1 - 2 * (q.x * q.x + q.y * q.y);
474 angles.x = std::atan2(sinr_cosp, cosr_cosp);
475
476 // pitch (y-axis rotation)
477 dfloat sinp = std::sqrt(1 + 2 * (q.w * q.y - q.x * q.z));
478 dfloat cosp = std::sqrt(1 - 2 * (q.w * q.y - q.x * q.z));
479 angles.y = 2 * std::atan2(sinp, cosp) - M_PI / 2;
480
481 // yaw (z-axis rotation)
482 dfloat siny_cosp = 2 * (q.w * q.z + q.x * q.y);
483 dfloat cosy_cosp = 1 - 2 * (q.y * q.y + q.z * q.z);
484 angles.z = std::atan2(siny_cosp, cosy_cosp);
485
486 return angles;
487}

528__host__ __device__
529dfloat4 compute_rotation_quart(dfloat3 v1, dfloat3 v2) {
530 v1 = vector_normalize(v1);
531 v2 = vector_normalize(v2);
532
533 dfloat dot = dot_product(v1, v2);
534
535 // Calculate the angle of rotation
536 dfloat angle_d2 = acosf(dot)*0.5_df;
537
538 // Calculate the axis of rotation
539 dfloat3 axis = cross_product(v1, v2);
540 axis = vector_normalize(axis);
541
542 dfloat4 q;
543 q.w = cosf(angle_d2 );
544 q.x = axis.x * sinf(angle_d2);
545 q.y = axis.y * sinf(angle_d2);
546 q.z = axis.z * sinf(angle_d2);
547
548 return q;
549}

594__host__ __device__
595dfloat mom_trilinear_interp(
596 dfloat x, dfloat y, dfloat z,
597 int mom,
598 dfloat *fMom)
599{
600 int i = (int)floor(x);
601 int j = (int)floor(y);
602 int k = (int)floor(z);
603
604 dfloat fx = x - i;
605 dfloat fy = y - j;
606 dfloat fz = z - k;
607
608 int i1 = i + 1;
609 int j1 = j + 1;
610 int k1 = k + 1;
611
612 #ifdef BC_X_PERIODIC
613 i = (i % NX + NX) % NX;
614 i1 = (i1 % NX + NX) % NX;
615 #elif defined(BC_X_WALL)
616 if (i < 0) i = 0;
617 if (i1 < 0) i1 = 0;
618 if (i >= NX) i = NX-1;
619 if (i1 >= NX) i1 = NX-1;
620 #endif
621
622 #ifdef BC_Y_PERIODIC
623 j = (j % NY + NY) % NY;
624 j1 = (j1 % NY + NY) % NY;
625 #elif defined(BC_Y_WALL)
626 if (j < 0) j = 0;
627 if (j1 < 0) j1 = 0;
628 if (j >= NY) j = NY-1;
629 if (j1 >= NY) j1 = NY-1;
630 #endif
631
632 #ifdef BC_Z_PERIODIC
633 k = (k % NZ + NZ) % NZ;
634 k1 = (k1 % NZ + NZ) % NZ;
635 #elif defined(BC_Z_WALL)
636 if (k < 0) k = 0;
637 if (k1 < 0) k1 = 0;
638 if (k >= NZ) k = NZ-1;
639 if (k1 >= NZ) k1 = NZ-1;
640 #endif
641
642 dfloat c000 = getMom(i, j, k, mom, fMom);
643 dfloat c100 = getMom(i1, j, k, mom, fMom);
644 dfloat c010 = getMom(i, j1, k, mom, fMom);
645 dfloat c110 = getMom(i1, j1, k, mom, fMom);
646
647 dfloat c001 = getMom(i, j, k1, mom, fMom);
648 dfloat c101 = getMom(i1, j, k1, mom, fMom);
649 dfloat c011 = getMom(i, j1, k1, mom, fMom);
650 dfloat c111 = getMom(i1, j1, k1, mom, fMom);
651
652 // trilinear interpolation
653 dfloat c00 = c000*(1-fx) + c100*fx;
654 dfloat c10 = c010*(1-fx) + c110*fx;
655
656 dfloat c01 = c001*(1-fx) + c101*fx;
657 dfloat c11 = c011*(1-fx) + c111*fx;
658
659 dfloat c0 = c00*(1-fy) + c10*fy;
660 dfloat c1 = c01*(1-fy) + c11*fy;
661
662 return c0*(1-fz) + c1*fz;
663}

667__host__ __device__
668dfloat cubic_interp(dfloat p0, dfloat p1, dfloat p2, dfloat p3, dfloat t) {
669 dfloat a = -0.5_df*p0 + 1.5_df*p1 - 1.5_df*p2 + 0.5_df*p3;
670 dfloat b = p0 - 2.5_df*p1 + 2.0_df*p2 - 0.5_df*p3;
671 dfloat c = -0.5_df*p0 + 0.5_df*p2;
672 dfloat d = p1;
673 return ((a*t + b)*t + c)*t + d;
674}

676__host__ __device__
677dfloat mom_tricubic_interp(dfloat x, dfloat y, dfloat z,
678 const int mom, dfloat *fMom) {
679 int i = (int)floor(x);
680 int j = (int)floor(y);
681 int k = (int)floor(z);
682
683 dfloat xi = x - i;
684 dfloat eta = y - j;
685 dfloat zeta = z - k;
686
687 dfloat F[4][4][4];
688
689 // collect 64 neighbors
690 for (int kk = 0; kk < 4; kk++) {
691 for (int jj = 0; jj < 4; jj++) {
692 for (int ii = 0; ii < 4; ii++) {
693 F[ii][jj][kk] = getMom(i + ii - 1, j + jj - 1, k + kk - 1, mom, fMom);
694 }
695 }
696 }
697
698 dfloat C[4][4]; // after x interpolation
699 dfloat D[4]; // after y interpolation
700
701 // interpolate in x for each row
702 for (int jj = 0; jj < 4; jj++) {
703 for (int kk = 0; kk < 4; kk++) {
704 C[jj][kk] = cubic_interp(F[0][jj][kk], F[1][jj][kk], F[2][jj][kk], F[3][jj][kk], xi);
705 }
706 }
707
708 // interpolate in y for each z
709 for (int kk = 0; kk < 4; kk++) {
710 D[kk] = cubic_interp(C[0][kk], C[1][kk], C[2][kk], C[3][kk], eta);
711 }
712
713 // interpolate in z
714 return cubic_interp(D[0], D[1], D[2], D[3], zeta);
715}
