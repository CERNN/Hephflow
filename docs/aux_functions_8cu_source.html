8void mean_rho(
9 dfloat *fMom,
10 size_t step,
11 dfloat *d_mean_rho
12){
13 dfloat* sumRho;
14 cudaMalloc((void**)&sumRho, NUM_BLOCK * sizeof(dfloat));
15
16 int nt_x = BLOCK_NX;
17 int nt_y = BLOCK_NY;
18 int nt_z = BLOCK_NZ;
19 int nb_x = NX / nt_x;
20 int nb_y = NY / nt_y;
21 int nb_z = NZ / nt_z;
22
23 sumReductionThread_rho << <dim3(NUM_BLOCK_X, NUM_BLOCK_Y, NUM_BLOCK_Z), dim3(BLOCK_NX, BLOCK_NY, BLOCK_NZ) >> > (fMom, sumRho);
24
25 nb_x = NUM_BLOCK_X;
26 nb_y = NUM_BLOCK_Y;
27 nb_z = NUM_BLOCK_Z;
28
29 int current_block_size = nb_x * nb_y * nb_z;
30
31 while (true) {
32 current_block_size = nb_x * nb_y * nb_z;
33 if (current_block_size <= BLOCK_LBM_SIZE) { // last reduction
34 sumReductionBlock << <1, dim3(nb_x, nb_y, nb_z) >> > (sumRho, sumRho);
35 break;
36 }
37 else {
38 nb_x = (nb_x < BLOCK_NX ? 1 : nb_x / BLOCK_NX);
39 nb_y = (nb_y < BLOCK_NY ? 1 : nb_y / BLOCK_NY);
40 nb_z = (nb_z < BLOCK_NZ ? 1 : nb_z / BLOCK_NZ);
41 if (nb_x * nb_y * nb_z * nt_x * nt_y * nt_z > current_block_size) {
42 if (nb_x > nb_y && nb_x > nb_z)
43 nt_x /= 2;
44 else if (nb_y > nb_x && nb_y > nb_z)
45 nt_y /= 2;
46 else
47 nt_z /= 2;
48 }
49 sumReductionBlock << <dim3(nb_x, nb_y, nb_z), dim3(nt_x, nt_y, nt_z) >> > (sumRho, sumRho);
50 }
51 }
52
53 checkCudaErrors(cudaDeviceSynchronize());
54 dfloat temp;
55
56 checkCudaErrors(cudaMemcpy(&temp, sumRho, sizeof(dfloat), cudaMemcpyDeviceToHost));
57 temp = (temp)/(NUMBER_LBM_NODES);
58 checkCudaErrors(cudaMemcpy(d_mean_rho, &temp, sizeof(dfloat), cudaMemcpyHostToDevice));

63 //strDataInfo <<"step,"<< step<< "," << temp;// << "," << mean_counter;
