151def readFileMacr3D(macrFilename,path):
152 """ Read the binary file and returns its content as a 3D matrix
153
154 Parameters
155 ----------
156 macrFilename : str
157 Filename of the macroscopic
158
159 Returns
160 -------
161 np.array([x, y, z])
162 Macroscopic array
163 """
164
165 info = getSimInfo(path)
166 if info['Prc'] == 'double':
167 prc = 'd'
168 elif info['Prc'] == 'float':
169 prc = 'f'
170 with open(macrFilename, "r") as f:
171 vec = np.fromfile(f, prc)
172 vec3D = np.reshape(vec, (info['NZ_TOTAL'], info['NY'], info['NX']), 'C')
173 return np.swapaxes(vec3D, 0, 2)
