6__host__
7void tracerSimulation(
8 ParticlesSoA *particles,
9 dfloat *fMom,
10 cudaStream_t streamParticles,
11 unsigned int step
12){
13
14 checkCudaErrors(cudaSetDevice(GPU_INDEX));
15 MethodRange range = particles->getMethodRange(TRACER);
16
17 int numTracerParticles = range.last - range.first + 1;
18 const unsigned int threadsNodes = 64;
19 const unsigned int gridNodes = (numTracerParticles + threadsNodes - 1) / threadsNodes;
20
21 if (particles == nullptr) {
22 printf("Error: particles is nullptr\n");
23 return;
24 }
25
26 checkCudaErrors(cudaStreamSynchronize(streamParticles));
27
28 if (range.first < 0 || range.last >= NUM_PARTICLES || range.first > range.last) {
29 printf("Error: Invalid range - first: %d, last: %d, NUM_PARTICLES: %d\n",
30 range.first, range.last, NUM_PARTICLES);
31 return;
32 }
33
34 ParticleCenter* pArray = particles->getPCenterArray();
35
36 tracer_positionUpdate<<<gridNodes, threadsNodes, 0, streamParticles>>>(
37 pArray, fMom,range.first,range.last,step
38 );
39
40 cudaError_t kernelError = cudaGetLastError();
41 if (kernelError != cudaSuccess) {
42 printf("Error in kernel tracer_positionUpdate: %s\n", cudaGetErrorString(kernelError));
43 }
44
45 checkCudaErrors(cudaStreamSynchronize(streamParticles));
46
47 #ifdef PARTICLE_TRACER_SAVE
48#pragma warning(push)
49#pragma warning(disable: 4804)
50 if(!(step%PARTICLE_TRACER_SAVE)){
51 checkCudaErrors(cudaMemcpy(h_particlePos, d_particlePos, sizeof(dfloat3)*NUM_PARTICLES, cudaMemcpyDeviceToHost));
52 saveParticleInfo(h_particlePos,step);
53 }
54#pragma warning(pop)
55 #endif //PARTICLE_TRACER_SAVE
56
57
58}

60__global__
61void tracer_positionUpdate(
62 ParticleCenter *pArray,
63 dfloat *fMom,
64 int firstIndex,
65 int lastIndex,
66 unsigned int step
67){
68 unsigned int localIdx = threadIdx.x + blockDim.x * blockIdx.x;
69 int globalIdx = firstIndex + localIdx;
70
71 if (globalIdx < firstIndex || globalIdx > lastIndex || globalIdx >= NUM_PARTICLES) {
72 return;
73 }
74
75 if (pArray == nullptr) {
76 printf("ERROR: particles is nullptr\n");
77 return;
78 }
79
80 if (globalIdx < firstIndex || globalIdx > lastIndex || globalIdx >= NUM_PARTICLES) {
81 return;
82 }
83
84 dfloat aux, aux1;
85 // Calculate stencils to use and the valid interval [xyz][idx]
86 dfloat stencilVal[3][P_DIST*2];
87
88 ParticleCenter* pc_i = &pArray[globalIdx];
89
90 dfloat3 pc_pos = pc_i->getPos();
91
92 dfloat pos[3] = {pc_pos.x,pc_pos.y,pc_pos.z};
93
94 const int posBase[3] = {
95 int(pos[0]) - (P_DIST) + 1,
96 int(pos[1]) - (P_DIST) + 1,
97 int(pos[2]) - (P_DIST) + 1
98 };
99
100 const int maxIdx[3] = {
101 #ifdef BC_X_WALL
102 ((posBase[0]+P_DIST*2-1) < (int)NX)? P_DIST*2-1 : ((int)NX-1-posBase[0])
103 #endif //BC_X_WALL
104 #ifdef BC_X_PERIODIC
105 P_DIST*2-1
106 #endif //BC_X_PERIODIC
107 ,
108 #ifdef BC_Y_WALL
109 ((posBase[1]+P_DIST*2-1) < (int)NY)? P_DIST*2-1 : ((int)NY-1-posBase[1])
110 #endif //BC_Y_WALL
111 #ifdef BC_Y_PERIODIC
112 P_DIST*2-1
113 #endif //BC_Y_PERIODIC
114 ,
115 #ifdef BC_Z_WALL
116 ((posBase[2]+P_DIST*2-1) < (int)NZ)? P_DIST*2-1 : ((int)NZ-1-posBase[2])
117 #endif //BC_Z_WALL
118 #ifdef BC_Z_PERIODIC
119 P_DIST*2-1
120 #endif //BC_Z_PERIODIC
121 };
122
123 const int minIdx[3] = {
124 #ifdef BC_X_WALL
125 (posBase[0] >= 0)? 0 : -posBase[0]
126 #endif //BC_X_WALL
127 #ifdef BC_X_PERIODIC
128 0
129 #endif //BC_X_PERIODIC
130 ,
131 #ifdef BC_Y_WALL
132 (posBase[1] >= 0)? 0 : -posBase[1]
133 #endif //BC_Y_WALL
134 #ifdef BC_Y_PERIODIC
135 0
136 #endif //BC_Y_PERIODIC
137 ,
138 #ifdef BC_Z_WALL
139 (posBase[2] >= 0)? 0 : -posBase[2]
140 #endif //BC_Z_WALL
141 #ifdef BC_Z_PERIODIC
142 0
143 #endif //BC_Z_PERIODIC
144 };
145
146 // Particle stencil out of the domain
147 if(maxIdx[0] <= 0 || maxIdx[1] <= 0 || maxIdx[2] <= 0)
148 return;
149 // Particle stencil out of the domain
150 if(minIdx[0] >= P_DIST*2 || minIdx[1] >= P_DIST*2 || minIdx[2] >= P_DIST*2)
151 return;
152
153 //calculate velocity interpolation stencil
154 for(int i = 0; i < 3; i++){
155 for(int j=minIdx[i]; j <= maxIdx[i]; j++){
156 stencilVal[i][j] = stencil(posBase[i]+j-(pos[i]));
157 }
158 }
159
160 dfloat uxVar = 0;
161 dfloat uyVar = 0;
162 dfloat uzVar = 0;
163
164 int posX, posY,posZ;
165 // Interpolation
166
167 for (int zk = minIdx[2]; zk <= maxIdx[2]; zk++) // z
168 {
169 for (int yj = minIdx[1]; yj <= maxIdx[1]; yj++) // y
170 {
171 aux1 = stencilVal[2][zk]*stencilVal[1][yj];
172 for (int xi = minIdx[0]; xi <= maxIdx[0]; xi++) // x
173 {
174 // Dirac delta (kernel)
175 aux = aux1 * stencilVal[0][xi];
176
177 posX =
178 #ifdef BC_X_WALL
179 posBase[0]+xi
180 #endif //IBM_BC_X_WALL
181 #ifdef BC_X_PERIODIC
182 (posBase[0]+xi + NX)%(NX)
183 #endif //BC_X_PERIODIC
184 ;
185 posY =
186 #ifdef BC_Y_WALL
187 posBase[1]+yj
188 #endif //BC_Y_WALL
189 #ifdef BC_Y_PERIODIC
190 (posBase[1]+yj + NY)%(NY)
191 #endif //BC_Y_PERIODIC
192 ;
193 posZ =
194 #ifdef BC_Z_WALL
195 posBase[2]+zk
196 #endif //BC_Z_WALL
197 #ifdef BC_Z_PERIODIC
198 (posBase[2]+zk + NZ)%(NZ)
199 #endif //BC_Z_PERIODIC
200 ;
201
202
203 uxVar += fMom[idxMom(posX%BLOCK_NX, posY%BLOCK_NY, posZ%BLOCK_NZ, M_UX_INDEX, posX/BLOCK_NX, posY/BLOCK_NY, posZ/BLOCK_NZ)] * aux;
204 uyVar += fMom[idxMom(posX%BLOCK_NX, posY%BLOCK_NY, posZ%BLOCK_NZ, M_UY_INDEX, posX/BLOCK_NX, posY/BLOCK_NY, posZ/BLOCK_NZ)] * aux;
205 uzVar += fMom[idxMom(posX%BLOCK_NX, posY%BLOCK_NY, posZ%BLOCK_NZ, M_UZ_INDEX, posX/BLOCK_NX, posY/BLOCK_NY, posZ/BLOCK_NZ)] * aux;
206 }
207 }
208 }
209
210 dfloat3 dVel = dfloat3(uxVar,uyVar,uzVar);// 0.01
211 //Update particle position
212 pc_i->setPos(pc_i->getPos() + dVel);
213
214 //AVOID THAT THE PARTICLES GO OUTSIDE OF THE DOMAIN
215 #ifdef BC_X_WALL
216 if (pc_i->getPosX() < 0) {
217 pc_i->setPosX(0.01_df);
218 }
219 if (pc_i->getPosX() > NX - 1) {
220 pc_i->setPosX(NX - 1.01_df);
221 }
222 #endif //BC_X_WALL
223
224 #ifdef BC_Y_WALL
225 if (pc_i->getPosY() < 0) {
226 pc_i->setPosY(0.01_df);
227 }
228 if (pc_i->getPosY() > NY - 1) {
229 pc_i->setPosY(NY - 1.01_df);
230 }
231 #endif //BC_Y_WALL
232
233 #ifdef BC_Z_WALL
234 if (pc_i->getPosZ() < 0) {
235 pc_i->setPosZ(0.01_df);
236 }
237 if (pc_i->getPosZ() > NZ - 1) {
238 pc_i->setPosZ(NZ - 1.01_df);
239 }
240 #endif //BC_Z_WALL
241
242}

245void tracer_saveParticleInfo(dfloat3 *h_particlePos, unsigned int step){
246 // Names of file to save particle info
247 std::string strFileParticleData = getVarFilename("particleData", step, ".csv");
248
249 // File to save particle info
250 std::ofstream outFileParticleData(strFileParticleData.c_str());
251
252 // String with all values as csv
253 std::ostringstream strValuesParticles("");
254 strValuesParticles << std::scientific;
255
256 // csv separator
257 std::string sep = ",";
258
259 std::string strColumnNames = "pIndex" + sep + "step" + sep;
260 strColumnNames += "pos_x" + sep + "pos_y" + sep + "pos_z";
261 strColumnNames += "\n";
262
263 for(int p = 0; p < NUM_PARTICLES; p++){
264 strValuesParticles << p << sep;
265 strValuesParticles << step << sep;
266 strValuesParticles << h_particlePos[p].x << sep << h_particlePos[p].y << sep << h_particlePos[p].z;
267 strValuesParticles << "\n";
268 }
269
270 outFileParticleData << strColumnNames << strValuesParticles.str();
271}
