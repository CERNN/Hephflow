10__device__
11dfloat3 computeNormalForce(const dfloat3& n, const dfloat3& G, dfloat displacement, dfloat stiffness, dfloat damping) {
12 dfloat f_kn = -stiffness * sqrt(abs(displacement*displacement*displacement));
13 return f_kn * n - damping * dot_product(G, n) * n * POW_FUNCTION(abs(displacement), 0.25);
14}
15
16__device__ dfloat3 computeTangentialForce(
17 dfloat3& tang_disp, // will be updated if slip occurs
18 const dfloat3& G_ct,
19 dfloat stiffness,
20 dfloat damping,
21 dfloat friction_coef,
22 dfloat f_n,
23 const dfloat3& t,
24 ParticleCenter* pc_i,
25 int tang_index,
26 int step
27) {
28 dfloat3 f_tang;
29 f_tang.x = - stiffness * tang_disp.x - damping * G_ct.x* POW_FUNCTION(abs(tang_disp.x) ,0.25);
30 f_tang.y = - stiffness * tang_disp.y - damping * G_ct.y* POW_FUNCTION(abs(tang_disp.y) ,0.25);
31 f_tang.z = - stiffness * tang_disp.z - damping * G_ct.z* POW_FUNCTION(abs(tang_disp.z) ,0.25);
32 dfloat mag = vector_length(f_tang);
33 if (mag > friction_coef * fabsf(f_n)) {
34 // CRITICAL: When slipping occurs, RESET displacement to zero, not subtract G_ct
35 // This prevents the accumulated elastic deformation from being applied
36 tang_disp = dfloat3{0.0, 0.0, 0.0};
37 updateTangentialDisplacement(pc_i->getCollision(), tang_index, tang_disp, step);
38 // Apply Coulomb (kinetic) friction
39 f_tang = -friction_coef * f_n * t;
40 }
41 return f_tang;
42}

44__device__ void accumulateForceAndTorque(ParticleCenter* pc_i, const dfloat3& f_dirs, const dfloat3& m_dirs) {
45 atomicAdd(&(pc_i->getFXatomic()), f_dirs.x);
46 atomicAdd(&(pc_i->getFYatomic()), f_dirs.y);
47 atomicAdd(&(pc_i->getFZatomic()), f_dirs.z);
48 atomicAdd(&(pc_i->getMXatomic()), m_dirs.x);
49 atomicAdd(&(pc_i->getMYatomic()), m_dirs.y);
50 atomicAdd(&(pc_i->getMZatomic()), m_dirs.z);
51}

59__device__
60int calculateWallIndex(const dfloat3 &n) {
61 // Calculate the index based on the normal vector
62
63 /*
64 n.x n.y n.z index
65 1 0 0 2
66 -1 0 0 4
67 0 1 0 5
68 0 -1 0 1
69 0 0 1 6
70 0 0 -1 0
71 0 0 0 3 -> external duct, since normal will be reduced to 0,0,0 when convert to integer
72 */
73
74 return 7 + (1 - (int)n.x) - 2 * (1 - (int)n.y) - 3 * (1 - (int)n.z);
75}

77__device__
78int getCollisionIndexByPartnerID(const CollisionData &collisionData, int partnerID, int currentTimeStep) {
79 for (int i = 7; i < MAX_ACTIVE_COLLISIONS ; i++) {
80 if (collisionData.getCollisionPartnerID(i) == partnerID &&
81 currentTimeStep - collisionData.getLastCollisionStep(i) <= 1) {
82 return i; // Found the collision index for a particle
83 }

85 // If no match is found, return -1
86 return -1;

89__device__
90int startCollision(CollisionData &collisionData, int partnerID, bool isWall, const dfloat3 &wallNormal, int currentTimeStep) {
91 int index = -1;
92 if (isWall) {
93 index = calculateWallIndex(wallNormal);
94 // Initialize wall collision data
95 collisionData.setTangentialDisplacement(index, {0.0, 0.0, 0.0});
96 collisionData.setLastCollisionStep(index, currentTimeStep);
97 } else {
98 for (int i = 7; i < MAX_ACTIVE_COLLISIONS; i++) {
99 if (collisionData.getLastCollisionStep(i) == -1) { // Check for an unused slot
100 collisionData.setCollisionPartnerID(i, partnerID);
101 collisionData.setTangentialDisplacement(i, {0.0, 0.0, 0.0});
102 collisionData.setLastCollisionStep(i, currentTimeStep);
103 index = i;
104 break;
105 }
106 }
107 }
108
109 return index;
110}

112__device__
113dfloat3 updateTangentialDisplacement(CollisionData &collisionData, int index, const dfloat3 &displacement, int currentTimeStep) {
114 collisionData.setTangentialDisplacement(index, (collisionData.getTangentialDisplacement(index) + displacement));
115 collisionData.setLastCollisionStep(index, currentTimeStep);
116 return collisionData.getTangentialDisplacement(index);
117}

119__device__
120void endCollision(CollisionData &collisionData, int index, int currentTimeStep) {
121 // Check if index is valid for wall collisions (0 to 6)
122 if (index >= 0 && index <= 6) {
123 collisionData.setLastCollisionStep(index, -1);
124 }
125 // Check if index is valid for particle collisions
126 else if (index >= 7 && index < MAX_ACTIVE_COLLISIONS ) {
127 if (currentTimeStep - collisionData.getLastCollisionStep(index) > 1) {
128 collisionData.setTangentialDisplacement(index, {0.0, 0.0, 0.0});
129 collisionData.setLastCollisionStep(index, -1); // Indicate the slot is available '
130 collisionData.setCollisionPartnerID(index, -1); // Optionally reset
131 }
132 }

136__device__
137dfloat3 getOrUpdateTangentialDisplacement(
138 ParticleCenter* pc_i,
139 int identifier, // wall index or partner ID
140 bool isWall, //true if wall, false if particle-particle collision
141 int step,
142 const dfloat3& G_ct,
143 const dfloat3& G_cn,
144 int& tang_index_out,
145 const dfloat3& wallNormal // only used for wall
146) {
147 dfloat3 tang_disp;
148 int tang_index = -1;
149
150 //get the collision index
151 if (isWall) {
152 tang_index = calculateWallIndex(wallNormal);
153 } else {
154 tang_index = getCollisionIndexByPartnerID(pc_i->getCollision(), identifier, step);
155 }
156
157 //check if there is a current collision
158 if (tang_index == -1) {
159 // NEW COLLISION: Initialize with zero displacement
160 tang_index = startCollision(pc_i->getCollision(), identifier, isWall, wallNormal, step);
161 tang_disp = dfloat3{0.0, 0.0, 0.0};
162 } else {
163 //check if the collision already exited in the past
164 if (step - pc_i->getCollision().getLastCollisionStep(tang_index) > 1) {
165 // COLLISION ENDED: Clean up old data
166 endCollision(pc_i->getCollision(), tang_index, step);
167 // START NEW COLLISION with fresh tracking
168 tang_index = startCollision(pc_i->getCollision(), identifier, isWall, wallNormal, step);
169 tang_disp = dfloat3{0.0, 0.0, 0.0};
170 } else {
171 // ONGOING COLLISION: Accumulate tangential displacement
172 tang_disp = updateTangentialDisplacement(pc_i->getCollision(), tang_index, G_ct, step);
173 }
174 }
175 //return the index by address
176 tang_index_out = tang_index;
177 //return the tangential displacement
178 return tang_disp;
179}

187__device__
188void sphereWallCollision(const CollisionContext& ctx, ParticleWallForces *d_pwForces){
189
190 ParticleCenter* pc_i = ctx.pc_i;
191 Wall wallData = ctx.wall;
192 dfloat displacement = ctx.displacement;
193 int step = ctx.step;
194
195 // Particle info
196 const dfloat m_i = pc_i->getVolume() * pc_i->getDensity();
197 const dfloat r_i = pc_i->getRadius();
198 const dfloat3 v_i = pc_i->getVel();
199 const dfloat3 w_i = pc_i->getW();
200
201 // Wall info
202 dfloat3 wall_speed = wallData.velocity;
203 dfloat3 n = wallData.normal;
204 n = n * -1.0f; //invert collision direction since is from sphere to wall
205
206 // Relative velocity
207 dfloat3 G = v_i - wall_speed;
208
209 //Collision constants
210 const dfloat STIFFNESS_NORMAL = SPHERE_WALL_STIFFNESS_NORMAL_CONST * sqrt(abs(r_i));
211 const dfloat STIFFNESS_TANGENTIAL = SPHERE_WALL_STIFFNESS_TANGENTIAL_CONST * sqrt(r_i) * sqrt(abs(displacement));
212 const dfloat DAMPING_NORMAL = SPHERE_WALL_DAMPING_CONST * sqrt(m_i * STIFFNESS_NORMAL);
213 const dfloat DAMPING_TANGENTIAL = SPHERE_WALL_DAMPING_CONST * sqrt(m_i * STIFFNESS_TANGENTIAL);
214
215 // Normal force
216 dfloat3 f_normal = computeNormalForce(n, G, displacement, STIFFNESS_NORMAL, DAMPING_NORMAL);
217 dfloat f_n = vector_length(f_normal);
218
219 // Relative tangential velocity
220 dfloat3 G_ct = G + r_i * cross_product(w_i, n) - dot_product(G, n) * n;
221 dfloat mag = vector_length(G_ct);
222 dfloat3 t = (mag != 0) ? (G_ct / mag) : dfloat3{0.0, 0.0, 0.0}; //tangential velocity vector
223
224
225 //retrive and update tangential displacement
226 int tang_index = calculateWallIndex(n); //wall can be directly computed
227 dfloat3 tang_disp = getOrUpdateTangentialDisplacement(pc_i, 0, true, step, G_ct, dot_product(G,n), tang_index, n);
228
229 // Compute tangential force
230 dfloat3 f_tang = computeTangentialForce(
231 tang_disp, G_ct, STIFFNESS_TANGENTIAL, DAMPING_TANGENTIAL,
232 PW_FRICTION_COEF, f_n, t, pc_i, tang_index, step
233 );
234
235 //Total forces and torque in the particle
236 dfloat3 f_dirs = f_normal + f_tang;
237 dfloat3 m_dirs = r_i * cross_product(n, f_tang);
238
239 //Save data in the particle information
240 accumulateForceAndTorque(pc_i, f_dirs, m_dirs);
241
242 // Force exerted by particle on wall
243 dfloat3 f_on_wall = -f_dirs;
244
245 // Magnitudes
246 dfloat Fn = vector_length(f_normal);
247 dfloat Ft = vector_length(f_tang);
248
249 // Select wall accumulator
250 ParticleWallForce* pwf = nullptr;
251
252 if (wallData.normal.y > 0) {
253 pwf = &d_pwForces->bottom;
254 } else if (wallData.normal.y < 0) {
255 pwf = &d_pwForces->top;
256 }
257
258 if (pwf) {
259 atomicAdd(&pwf->Fx, f_on_wall.x);
260 atomicAdd(&pwf->Fy, f_on_wall.y);
261 atomicAdd(&pwf->Fz, f_on_wall.z);
262
263 atomicAdd(&pwf->Fn, Fn);
264 atomicAdd(&pwf->Ft, Ft);
265
266 atomicAdd(&pwf->nContacts, 1);
267 }
268}

275 dfloat displacement = ctx.displacement;
276 int step = ctx.step;
277 dfloat3 endpoint = ctx.contactPoint;

279 const dfloat3 center_pos = pc_i->getPos(); // Particle center position
280 const dfloat3 cap_pos = endpoint; // Capsule cap position
281 const dfloat m_i = pc_i->getVolume() * pc_i->getDensity();
282 const dfloat r_i = pc_i->getRadius();
283 const dfloat3 v_i = pc_i->getVel(); // Center of mass velocity

288
289 // Relative velocity
290 dfloat3 rri = cap_pos - center_pos;
291 dfloat3 G = (v_i + cross_product(w_i,r_i*n + rri)) - (wall_speed);
292 dfloat3 G_cn = dot_product(G,n) * n;
293 dfloat3 G_ct = G - G_cn;
294
295 //Collision constants
296 const dfloat STIFFNESS_NORMAL = SPHERE_WALL_STIFFNESS_NORMAL_CONST * sqrt(abs(r_i));
297 const dfloat STIFFNESS_TANGENTIAL = SPHERE_WALL_STIFFNESS_TANGENTIAL_CONST * sqrt(r_i) * sqrt(abs(displacement));
298 const dfloat DAMPING_NORMAL = SPHERE_WALL_DAMPING_CONST * sqrt(m_i * STIFFNESS_NORMAL);
299 const dfloat DAMPING_TANGENTIAL = SPHERE_WALL_DAMPING_CONST * sqrt(m_i * STIFFNESS_TANGENTIAL);
300
301
302 // Normal force
303 dfloat3 f_normal = computeNormalForce(n, G, displacement, STIFFNESS_NORMAL, DAMPING_NORMAL);
304 dfloat f_n = vector_length(f_normal);
305
306 // Relative tangential velocity
307 dfloat mag = vector_length(G_ct);
308 dfloat3 t = (mag != 0) ? (G_ct / mag) : dfloat3{0.0, 0.0, 0.0}; // Tangential velocity vector
309
310 // Tangential displacement tracking
311 //retrive and update tangential displacement
312 int tang_index = calculateWallIndex(n); //wall can be directly computed
313 dfloat3 tang_disp = getOrUpdateTangentialDisplacement(pc_i, 0, true, step, G_ct, G_cn, tang_index, n);
314
315
316 // Compute tangential force
317 dfloat3 f_tang = computeTangentialForce(
318 tang_disp, G_ct, STIFFNESS_TANGENTIAL, DAMPING_TANGENTIAL,
319 PW_FRICTION_COEF, f_n, t, pc_i, tang_index, step
320 );
321
322 //Total forces and torque in the particle
323 dfloat3 f_dirs = f_normal + f_tang;
324 dfloat3 m_dirs = cross_product((n * r_i) + rri, f_dirs);
325
326 //Save data in the particle information
327 accumulateForceAndTorque(pc_i, f_dirs, m_dirs);
328}

467// -------------------------- SPHERE COLLISION --------------------------------
468__device__
469void sphereSphereCollision(const CollisionContext& ctx){
470 ParticleCenter* pc_i = ctx.pc_i;
471 ParticleCenter* pc_j = ctx.pc_j;
472 int step = ctx.step;
473 dfloat displacement = ctx.displacement;
474 int partnerID = ctx.partnerID;
475 // Particle i info
476 const dfloat r_i = pc_i->getRadius();
477 const dfloat m_i = pc_i->getVolume() * pc_i->getDensity();
478 const dfloat3 v_i = pc_i->getVel();
479 const dfloat3 w_i = pc_i->getW();
480 // Particle j info
481 const dfloat r_j = pc_j->getRadius();
482 const dfloat m_j = pc_j->getVolume() * pc_j->getDensity();
483 const dfloat3 v_j = pc_j->getVel();
484 const dfloat3 w_j = pc_j->getW();
485
486 // Use normal direction from context (set in wall.normal)
487 const dfloat3 n = ctx.wall.normal;
488 // Relative velocity
489 dfloat3 G = v_i - v_j;
490
491 // Hertz contact theory
492 dfloat effective_radius = 1.0 / ((r_i + r_j) / (r_i * r_j));
493 dfloat effective_mass = 1.0 / ((m_i + m_j) / (m_i * m_j));
494 const dfloat STIFFNESS_NORMAL = SPHERE_SPHERE_STIFFNESS_NORMAL_CONST * sqrt(effective_radius);
495 const dfloat STIFFNESS_TANGENTIAL = SPHERE_SPHERE_STIFFNESS_TANGENTIAL_CONST * sqrt(effective_radius) * sqrt(abs(displacement));
496 const dfloat DAMPING_NORMAL = SPHERE_SPHERE_DAMPING_CONST * sqrt(effective_mass * STIFFNESS_NORMAL);
497 const dfloat DAMPING_TANGENTIAL = SPHERE_SPHERE_DAMPING_CONST * sqrt(effective_mass * STIFFNESS_TANGENTIAL);
498
499 // Normal force
500 dfloat3 f_normal = computeNormalForce(n, G, displacement, STIFFNESS_NORMAL, DAMPING_NORMAL);
501 dfloat f_n = vector_length(f_normal);
502
503 // Relative tangential velocity
504 dfloat3 G_ct = G + r_i * cross_product(w_i, n) + r_j * cross_product(w_j, n) - dot_product(G, n) * n;
505 dfloat mag = vector_length(G_ct);
506 dfloat3 t = (mag != 0) ? (G_ct / mag) : dfloat3{0.0, 0.0, 0.0}; //tangential velocity vector
507
508 int tang_index = -1;
509 dfloat3 tang_disp = getOrUpdateTangentialDisplacement(pc_i, partnerID, false, step, G_ct, G, tang_index, dfloat3(0, 0, 0));
510
511 // Compute tangential force
512 dfloat3 f_tang = computeTangentialForce(
513 tang_disp, G_ct, STIFFNESS_TANGENTIAL, DAMPING_TANGENTIAL,
514 PW_FRICTION_COEF, f_n, t, pc_i, tang_index, step
515 );
516
517 // Final force results
518 dfloat3 f_dirs = f_normal + f_tang;
519 dfloat3 m_dirs_i = r_i * cross_product(n, f_tang);
520 dfloat3 m_dirs_j = r_j * cross_product(n, f_tang);
521
522 //Save data in the particle information
523 accumulateForceAndTorque(pc_i, -f_dirs, m_dirs_i);
524 accumulateForceAndTorque(pc_j, f_dirs, m_dirs_j);
525}

554 dfloat3 G = v_i - v_j;
555
556 // Hertz contact theory
557 dfloat effective_radius = 1.0 / ((r_i + r_j) / (r_i * r_j));
558 dfloat effective_mass = 1.0 / ((m_i + m_j) / (m_i * m_j));
559 const dfloat STIFFNESS_NORMAL = SPHERE_SPHERE_STIFFNESS_NORMAL_CONST * sqrt(effective_radius);
560 const dfloat STIFFNESS_TANGENTIAL = SPHERE_SPHERE_STIFFNESS_TANGENTIAL_CONST * sqrt(effective_radius) * sqrt(abs(displacement));
561 const dfloat DAMPING_NORMAL = SPHERE_SPHERE_DAMPING_CONST * sqrt(effective_mass * STIFFNESS_NORMAL);
562 const dfloat DAMPING_TANGENTIAL = SPHERE_SPHERE_DAMPING_CONST * sqrt(effective_mass * STIFFNESS_TANGENTIAL);
563
564 // Normal force
565 dfloat3 f_normal = computeNormalForce(n, G, displacement, STIFFNESS_NORMAL, DAMPING_NORMAL);
566 dfloat f_n = vector_length(f_normal);
567
568 // Relative tangential velocity
569 dfloat3 G_ct = G + r_i * cross_product(w_i, n) + r_j * cross_product(w_j, n) - dot_product(G, n) * n;
570 dfloat mag = vector_length(G_ct);
571 dfloat3 t = (mag != 0) ? (G_ct / mag) : dfloat3{0.0, 0.0, 0.0};
572
573 int tang_index = -1;
574 dfloat3 tang_disp = getOrUpdateTangentialDisplacement(pc_i, partnerID, false, step, G_ct, dot_product(G,n), tang_index, dfloat3(0, 0, 0));
575
576 // Compute tangential force
577 dfloat3 f_tang = computeTangentialForce(
578 tang_disp, G_ct, STIFFNESS_TANGENTIAL, DAMPING_TANGENTIAL,
579 PP_FRICTION_COEF, f_n, t, pc_i, tang_index, step
580 );
581
582 // Final force results
583 dfloat3 f_dirs = f_normal + f_tang;
584 dfloat3 m_dirs_i = cross_product((pos_i - pos_c_i) + (-n * r_i), -f_dirs);
585 dfloat3 m_dirs_j = cross_product((pos_j - pos_c_j) + (n * r_j), f_dirs);
586
587 //Save data in the particle information
588 accumulateForceAndTorque(pc_i, -f_dirs, m_dirs_i);
589 accumulateForceAndTorque(pc_j, f_dirs, m_dirs_j);
590}

741dfloat3 ellipsoid_intersection(ParticleCenter* pc_i, dfloat R[3][3], dfloat3 line_origin, dfloat3 line_dir, dfloat3 translation) {
742 dfloat3 p0, p0_rotated, d_rotated;
743 dfloat A, B, C;
744 dfloat DELTA;
745 dfloat3 t;
746
747 dfloat3 pos = pc_i->getPos();
748 dfloat3 center = pos + translation;
749
750 dfloat a_axis = vector_length(pc_i->getSemiAxis1() - pos);
751 dfloat b_axis = vector_length(pc_i->getSemiAxis2() - pos);
752 dfloat c_axis = vector_length(pc_i->getSemiAxis3() - pos);
753
754 dfloat inva2 = 1.0f / (a_axis * a_axis);
755 dfloat invb2 = 1.0f / (b_axis * b_axis);
756 dfloat invc2 = 1.0f / (c_axis * c_axis);
757
758 p0 = line_origin - center;
759
760 // Apply rotation to the line origin
761 p0_rotated.x = R[0][0] * p0.x + R[0][1] * p0.y + R[0][2] * p0.z;
762 p0_rotated.y = R[1][0] * p0.x + R[1][1] * p0.y + R[1][2] * p0.z;
763 p0_rotated.z = R[2][0] * p0.x + R[2][1] * p0.y + R[2][2] * p0.z;
764
765 // Transform the line direction into the ellipsoid's coordinate system
766 d_rotated.x = R[0][0] * line_dir.x + R[0][1] * line_dir.y + R[0][2] * line_dir.z;
767 d_rotated.y = R[1][0] * line_dir.x + R[1][1] * line_dir.y + R[1][2] * line_dir.z;
768 d_rotated.z = R[2][0] * line_dir.x + R[2][1] * line_dir.y + R[2][2] * line_dir.z;
769
770 // Precompute scaled components
771 dfloat px_inva2 = p0_rotated.x * inva2;
772 dfloat py_invb2 = p0_rotated.y * invb2;
773 dfloat pz_invc2 = p0_rotated.z * invc2;
774
775 dfloat dx_inva2 = d_rotated.x * inva2;
776 dfloat dy_invb2 = d_rotated.y * invb2;
777 dfloat dz_invc2 = d_rotated.z * invc2;
778
779 // Ellipsoid equation coefficients (in rotated space)
780 A = d_rotated.x * dx_inva2 + d_rotated.y * dy_invb2 + d_rotated.z * dz_invc2;
781 B = 2.0f * (p0_rotated.x * dx_inva2 + p0_rotated.y * dy_invb2 + p0_rotated.z * dz_invc2);
782 C = p0_rotated.x * px_inva2 + p0_rotated.y * py_invb2 + p0_rotated.z * pz_invc2 - 1.0f;
783
784 DELTA = B * B - 4.0f * A * C;
785
786 dfloat sqrt_delta = sqrtf(DELTA);
787 dfloat denom = 2.0f * A;
788 t = dfloat3((-B + sqrt_delta) / denom, (-B - sqrt_delta) / denom, 0.0f);
789
790 return t;
791}

794dfloat3 ellipsoid_normal(ParticleCenter* pc_i, dfloat R[3][3], dfloat3 point, dfloat radius[1], dfloat3 translation) {
795 dfloat3 local_point;
796 dfloat3 grad_local;
797 dfloat3 normal;
798 dfloat norm;
799
800 dfloat3 pos = pc_i->getPos();
801 dfloat3 center = pos + translation;
802
803 dfloat a_axis = vector_length(pc_i->getSemiAxis1() - pos);
804 dfloat b_axis = vector_length(pc_i->getSemiAxis2() - pos);
805 dfloat c_axis = vector_length(pc_i->getSemiAxis3() - pos);
806
807 dfloat a2 = a_axis * a_axis;
808 dfloat b2 = b_axis * b_axis;
809 dfloat c2 = c_axis * c_axis;
810
811 dfloat inva2 = 1.0f / a2;
812 dfloat invb2 = 1.0f / b2;
813 dfloat invc2 = 1.0f / c2;
814
815 dfloat a4 = a2 * a2;
816 dfloat b4 = b2 * b2;
817 dfloat c4 = c2 * c2;
818
819 dfloat3 diff = point - center;
820
821 // Transform the point into the ellipsoid's local coordinates
822 local_point.x = R[0][0] * diff.x + R[0][1] * diff.y + R[0][2] * diff.z;
823 local_point.y = R[1][0] * diff.x + R[1][1] * diff.y + R[1][2] * diff.z;
824 local_point.z = R[2][0] * diff.x + R[2][1] * diff.y + R[2][2] * diff.z;
825
826 dfloat x2 = local_point.x * local_point.x;
827 dfloat y2 = local_point.y * local_point.y;
828 dfloat z2 = local_point.z * local_point.z;
829
830 dfloat a4b4 = a4 * b4;
831 dfloat a4c4 = a4 * c4;
832 dfloat b4c4 = b4 * c4;
833
834 dfloat inner = a4b4 * z2 + a4c4 * y2 + b4c4 * x2;
835 dfloat abc = a_axis * b_axis * c_axis;
836 dfloat denom = a4 * b4 * c4;
837 radius[0] = abc * inner / denom;
838
839 // Compute the gradient in local coordinates
840 grad_local.x = 2.0f * local_point.x * inva2;
841 grad_local.y = 2.0f * local_point.y * invb2;
842 grad_local.z = 2.0f * local_point.z * invc2;
843
844 // Transform the gradient back to global coordinates
845 normal.x = R[0][0] * grad_local.x + R[1][0] * grad_local.y + R[2][0] * grad_local.z;
846 normal.y = R[0][1] * grad_local.x + R[1][1] * grad_local.y + R[2][1] * grad_local.z;
847 normal.z = R[0][2] * grad_local.x + R[1][2] * grad_local.y + R[2][2] * grad_local.z;
848
849 // Normalize the normal vector
850 dfloat norm2 = normal.x * normal.x + normal.y * normal.y + normal.z * normal.z;
851 if (norm2 > 0.0f) { // Avoid division by zero
852 norm = sqrtf(norm2);
853 dfloat inv_norm = 1.0f / norm;
854 normal.x *= inv_norm;
855 normal.y *= inv_norm;
856 normal.z *= inv_norm;
857 }
858
859 return normal;
860}

863dfloat ellipsoidEllipsoidCollisionDistance(ParticleCenter* pc_i, ParticleCenter* pc_j, dfloat3 contactPoint1[1], dfloat3 contactPoint2[1], dfloat cr1[1], dfloat cr2[1], dfloat3 translation, unsigned int step) {
864 dfloat R1[3][3];
865 dfloat R2[3][3];
866 dfloat3 sphere_center1, sphere_center2;
867 dfloat3 closest_point1, closest_point2;
868 dfloat3 new_sphere_center1, new_sphere_center2;
869
870 dfloat3 pos_i = pc_i->getPos();
871 dfloat3 pos_j = pc_j->getPos();
872 dfloat3 center_j = pos_j + translation;
873
874 // Precompute semi-axis vectors and lengths
875 dfloat3 vec_a1 = pc_i->getSemiAxis1() - pos_i;
876 dfloat3 vec_b1 = pc_i->getSemiAxis2() - pos_i;
877 dfloat3 vec_c1 = pc_i->getSemiAxis3() - pos_i;
878 dfloat a1 = vector_length(vec_a1);
879 dfloat b1 = vector_length(vec_b1);
880 dfloat c1 = vector_length(vec_c1);
881
882 dfloat3 vec_a2 = pc_j->getSemiAxis1() - pos_j;
883 dfloat3 vec_b2 = pc_j->getSemiAxis2() - pos_j;
884 dfloat3 vec_c2 = pc_j->getSemiAxis3() - pos_j;
885 dfloat a2 = vector_length(vec_a2);
886 dfloat b2 = vector_length(vec_b2);
887 dfloat c2 = vector_length(vec_c2);
888
889 // Compute unit vectors for rotation matrices
890 dfloat3 unit_a1 = vec_a1 / a1;
891 dfloat3 unit_b1 = vec_b1 / b1;
892 dfloat3 unit_c1 = vec_c1 / c1;
893
894 dfloat3 unit_a2 = vec_a2 / a2;
895 dfloat3 unit_b2 = vec_b2 / b2;
896 dfloat3 unit_c2 = vec_c2 / c2;
897
898 // Obtain rotation matrices
899 rotationMatrixFromVectors(unit_a1, unit_b1, unit_c1, R1);
900 rotationMatrixFromVectors(unit_a2, unit_b2, unit_c2, R2);
901
902 dfloat3 dir = center_j - pos_i;
903
904 dfloat3 t1 = ellipsoid_intersection(pc_i, R1, pos_i, dir, dfloat3(0.0f, 0.0f, 0.0f));
905 dfloat3 t2 = ellipsoid_intersection(pc_j, R2, pos_i, dir, translation);
906
907 computeContactPoints(pos_i, dir, t1, t2, &closest_point1, &closest_point2);
908
909 dfloat r = 3.0; //TODO FIND A BETTER WAY TO GET THE BEST RADIUS FOR THIS DETECTION
910
911
912 dfloat3 normal1 = ellipsoid_normal(pc_i, R1, closest_point1, cr1, dfloat3(0.0f, 0.0f, 0.0f));
913 dfloat3 normal2 = ellipsoid_normal(pc_j, R2, closest_point2, cr2, translation);
914
915
916
917 sphere_center1 = closest_point1 - r * normal1;
918 sphere_center2 = closest_point2 - r * normal2;
919
920 // Iteration loop with squared tolerance
921 const int max_iters = 20;
922 const dfloat tolerance_sq = 1e-6f;
923 for (int i = 0; i < max_iters; ++i) {
924 dir = sphere_center2 - sphere_center1;
925
926 t1 = ellipsoid_intersection(pc_i, R1, sphere_center1, dir, dfloat3(0.0f, 0.0f, 0.0f));
927 t2 = ellipsoid_intersection(pc_j, R2, sphere_center1, dir, translation);
928
929 computeContactPoints(sphere_center1, dir, t1, t2, &closest_point1, &closest_point2);
930
931 normal1 = ellipsoid_normal(pc_i, R1, closest_point1, cr1, dfloat3(0.0f, 0.0f, 0.0f));
932 normal2 = ellipsoid_normal(pc_j, R2, closest_point2, cr2, translation);
933
934 new_sphere_center1 = closest_point1 - r * normal1;
935 new_sphere_center2 = closest_point2 - r * normal2;
936
937 dfloat3 diff1 = new_sphere_center1 - sphere_center1;
938 dfloat3 diff2 = new_sphere_center2 - sphere_center2;
939 dfloat error_sq = (diff1.x * diff1.x + diff1.y * diff1.y + diff1.z * diff1.z) +
940 (diff2.x * diff2.x + diff2.y * diff2.y + diff2.z * diff2.z);
941
942 if (error_sq < tolerance_sq) {
943 break;
944 }
945
946 sphere_center1 = new_sphere_center1;
947 sphere_center2 = new_sphere_center2;
948 }
949
950 contactPoint1[0] = closest_point1;
951 contactPoint2[0] = closest_point2;
952
953 dfloat3 center_diff = sphere_center2 - sphere_center1;
954 return vector_length(center_diff) - 2.0f * r;
955}
956__device__
957void computeContactPoints(dfloat3 pos_i, dfloat3 dir, dfloat3 t1, dfloat3 t2, dfloat3 contactPoint1[1], dfloat3 contactPoint2[1]) {
958 // Calculate the squared distances between all four pairs of t values.
959 // The vector_length(dir) is a constant scaling factor, so we can
960 // omit it for the comparison and re-introduce it if needed later.
961 dfloat dist_sq00 = (t1.x - t2.x) * (t1.x - t2.x);
962 dfloat dist_sq01 = (t1.x - t2.y) * (t1.x - t2.y);
963 dfloat dist_sq10 = (t1.y - t2.x) * (t1.y - t2.x);
964 dfloat dist_sq11 = (t1.y - t2.y) * (t1.y - t2.y);
965
966 // Find the minimum squared distance and its indices
967 dfloat min_dist_sq = dist_sq00;
968 int i_min = 0;
969 int j_min = 0;
970
971 if (dist_sq01 < min_dist_sq) {
972 min_dist_sq = dist_sq01;
973 i_min = 0;
974 j_min = 1;
975 }
976 if (dist_sq10 < min_dist_sq) {
977 min_dist_sq = dist_sq10;
978 i_min = 1;
979 j_min = 0;
980 }
981 if (dist_sq11 < min_dist_sq) {
982 min_dist_sq = dist_sq11;
983 i_min = 1;
984 j_min = 1;
985 }
986
987 // Select the correct t values based on the minimum distance indices.
988 dfloat ta = (i_min == 0) ? t1.x : t1.y;
989 dfloat tb = (j_min == 0) ? t2.x : t2.y;
990
991 // Compute the contact points just once.
992 contactPoint1[0] = pos_i + ta * dir;
993 contactPoint2[0] = pos_i + tb * dir;
994}
