135void IbmNodesSoA::copyNodesFromParticle(Particle *p, unsigned int pCenterIdx, ParticleCenter* pArray, unsigned int n_gpu)
136{
137 const int baseIdx = this->numNodes;
138 int nodesAdded = 0;
139 IbmNodes* node = p->getNode();
140
141 // Get particle center position for computing relative offsets
142 dfloat3 pc_pos = pArray[pCenterIdx].getPos();
143
144 for (int i = 0; i < p->getNumNodes(); i++)
145 {
146
147 if(!is_inside_gpu(node[i].getPos(), n_gpu))
148 continue;
149
150 this->particleCenterIdx[nodesAdded+baseIdx] = pCenterIdx;
151
152 this->pos.copyValuesFromFloat3(node[i].getPos(), nodesAdded+baseIdx);
153
154 // Compute and store relative offset (node position - particle center position)
155 // This immutable reference prevents error accumulation
156 dfloat3 node_pos = node[i].getPos();
157 dfloat3 relative_pos = dfloat3(
158 node_pos.x - pc_pos.x,
159 node_pos.y - pc_pos.y,
160 node_pos.z - pc_pos.z
161 );
162 this->originalRelativePos.copyValuesFromFloat3(relative_pos, nodesAdded+baseIdx);
163
164 this->vel.copyValuesFromFloat3(node[i].getVel(), nodesAdded+baseIdx);
165 this->vel_old.copyValuesFromFloat3(node[i].getVelOld(), nodesAdded+baseIdx);
166 this->f.copyValuesFromFloat3(node[i].getF(), nodesAdded+baseIdx);
167 this->deltaF.copyValuesFromFloat3(node[i].getDeltaF(), nodesAdded+baseIdx);
168 this->S[nodesAdded+baseIdx] = node[i].getS();
169 nodesAdded += 1;
170 }
171
172 this->numNodes += nodesAdded;
173}
