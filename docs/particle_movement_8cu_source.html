190void updateParticlePosition(

192 unsigned int step)
193{
194 unsigned int localIdx = threadIdx.x + blockDim.x * blockIdx.x;
195 int globalIdx = localIdx;
196
197 if (globalIdx >= NUM_PARTICLES) {
198 return;
199 }
200
201 if (pArray == nullptr) {
202 printf("ERROR: particles is nullptr\n");
203 return;
204 }
205
206 ParticleCenter* pc_i = &pArray[globalIdx];
207
208 if(!pc_i->getMovable())
209 return;
210
211 #ifdef PARTICLE_DEBUG
212 printf("updateParticlePosition 1 pos x: %e y: %e z: %e\n",pc_i->getPosX(),pc_i->getPosY(),pc_i->getPosZ());
213 printf("updateParticlePosition 1 vel x: %e y: %e z: %e\n",pc_i->getVel().x,pc_i->getVel().y,pc_i->getVel().z);
214 printf("updateParticlePosition 1 w x: %e y: %e z: %e\n",pc_i->getWX(),pc_i->getWY(),pc_i->getWZ());
215 #endif //PARTICLE_DEBUG
216
217 #ifdef BC_X_WALL
218 pc_i->setPosX(pc_i->getPosX() + (pc_i->getVelX() + pc_i->getVelOldX())/2);
219 #endif //BC_X_WALL
220 #ifdef BC_X_PERIODIC
221 dfloat dx = (pc_i->getVelX() + pc_i->getVelOldX())/2;
222 dfloat new_x = pc_i->getPosX() + dx;
223 dfloat mod_x = std::fmod(new_x, (dfloat)NX);
224 pc_i->setPosX((mod_x < 0) ? mod_x + (dfloat)NX : mod_x);
225 #endif //BC_X_PERIODIC
226
227 #ifdef BC_Y_WALL
228 pc_i->setPosY(pc_i->getPosY() + (pc_i->getVelY() + pc_i->getVelOldY())/2);
229 #endif //BC_Y_WALL
230 #ifdef BC_Y_PERIODIC
231 dfloat dy = (pc_i->getVelY() + pc_i->getVelOldY())/2;
232 dfloat new_y = pc_i->getPosY() + dy;
233 dfloat mod_y = std::fmod(new_y, (dfloat)NY);
234 pc_i->setPosY((mod_y < 0) ? mod_y + (dfloat)NY : mod_y);
235 #endif //BC_Y_PERIODIC
236
237 #ifdef BC_Z_WALL
238 pc_i->setPosZ(pc_i->getPosZ() + (pc_i->getVelZ() + pc_i->getVelOldZ())/2);
239 #endif //BC_Z_WALL
240 #ifdef BC_Z_PERIODIC
241 dfloat dz = (pc_i->getVelZ() + pc_i->getVelOldZ())/2;
242 dfloat new_z = pc_i->getPosZ() + dz;
243 dfloat mod_z = std::fmod(new_z, (dfloat)NZ_TOTAL);
244 pc_i->setPosZ((mod_z < 0) ? mod_z + (dfloat)NZ_TOTAL : mod_z);
245 #endif //BC_Z_PERIODIC
246
247 //Compute angular velocity
248 pc_i->setW_avg((pc_i->getW() + pc_i->getW_old())/2);
249
250 //update angular position
251 pc_i->setW_pos(pc_i->getW_pos() + pc_i->getW_avg());
252
253 #ifdef PARTICLE_DEBUG
254 printf("updateParticlePosition 2 pos x: %e y: %e z: %e\n",pc_i->getPosX(),pc_i->getPosY(),pc_i->getPosZ());
255 printf("updateParticlePosition 2 vel x: %e y: %e z: %e\n",pc_i->getVel().x,pc_i->getVel().y,pc_i->getVel().z);
256 printf("updateParticlePosition 2 w x: %e y: %e z: %e\n",pc_i->getWX(),pc_i->getWY(),pc_i->getWZ());
257 #endif //PARTICLE_DEBUG
258
259
260 const dfloat w_norm = sqrt((pc_i->getWAvgX() * pc_i->getWAvgX())
261 + (pc_i->getWAvgY() * pc_i->getWAvgY())
262 + (pc_i->getWAvgZ() * pc_i->getWAvgZ()));
263 dfloat3 axis = {0,0,0};
264 if (w_norm > 1e-8) {
265 axis.x = pc_i->getWAvgX() / w_norm;
266 axis.y = pc_i->getWAvgY() / w_norm;
267 axis.z = pc_i->getWAvgZ() / w_norm;
268 }
269 dfloat angle = w_norm;
270 dfloat4 q = axis_angle_to_quart(axis, angle);
271
272 dfloat4 q_cumulative = pc_i->getQ_cumulative_rot();
273 q_cumulative = quart_multiplication(q, q_cumulative);
274 q_cumulative = quart_normalize(q_cumulative);
275
276
277 // Store updated cumulative rotation back to particle center
278 pc_i->setQ_cumulative_rot(q_cumulative);
279
280 dfloat3 pos_old = pc_i->getPos_old();
281
282 dfloat3 pos_new = pc_i->getPos();
283
284 pc_i->setDx(pos_new - pos_old);
285
286 // Update semi-axes using cumulative rotation and original offsets
287 // This avoids error accumulation from incremental updates
288 pc_i->setSemiAxis1(updateSemiAxis(pc_i->getSemiAxis1Original(), pos_new, q_cumulative));
289 pc_i->setSemiAxis2(updateSemiAxis(pc_i->getSemiAxis2Original(), pos_new, q_cumulative));
290 pc_i->setSemiAxis3(updateSemiAxis(pc_i->getSemiAxis3Original(), pos_new, q_cumulative));
291}
