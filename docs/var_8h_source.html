150constexpr dfloat constexprLnHelper(dfloat y, int n, dfloat sum) {
151 if (n > 10) {
152 return sum;
153 }
154 dfloat term = y;
155 for (int i = 0; i < n - 1; ++i) {
156 term *= y * y;
157 }
158 return constexprLnHelper(y, n + 1, sum + term / (2.0_df * n - 1.0_df));
159}

161constexpr dfloat constexprLn(dfloat x) {
162 if (x <= 0.0_df) {
163 return std::numeric_limits<dfloat>::quiet_NaN();
164 }
165 dfloat y = (x - 1.0_df) / (x + 1.0_df);
166 return 2.0_df * constexprLnHelper(y, 1, 0.0_df);
167}

174// Helper function to get the closest power of 2 under or equal to `n`
175constexpr int closestPowerOfTwo(int n) {
176 int power = 1;
177 while (power * 2 <= n) {
178 power *= 2;

187// Compute optimal block dimensions
188constexpr BlockDim findOptimalBlockDimensions(size_t maxElements) {
189 int bestX = 1, bestY = 1, bestZ = 1;
190 int closestVolume = 0;
191 float bestForm = 0.0_df;
192
193 // Iterate over powers of 2 up to `maxElements` to find optimal dimensions
194 for (int x = closestPowerOfTwo(maxElements); x >= 1; x /= 2) {
195 for (int y = closestPowerOfTwo(maxElements / x); y >= 1; y /= 2) {
196 for (int z = closestPowerOfTwo(maxElements / (x * y)); z >= 1; z /= 2) {
197 if (x * y * z <= maxElements) {
198 int volume = x * y * z;
199 float form = 1.0_df/(1.0_df/x + 1.0_df/y + 1.0_df/z);
200 if (volume > closestVolume) {
201 bestX = x;
202 bestY = y;
203 bestZ = z;
204 closestVolume = volume;
205 bestForm = form;
206 } else if (volume == closestVolume && form > bestForm) {
207 bestX = x;
208 bestY = y;
209 bestZ = z;
210 bestForm = form;
211 }
212 }
213 }
214 }
215 }
216 return {bestX, bestY, bestZ};
217}
